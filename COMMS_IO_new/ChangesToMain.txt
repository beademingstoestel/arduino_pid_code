void doWatchdog(void) {
  if (millis() - lastWatchdogTime > Watchdog) {
    setAlarmState(2);
	resetComm();
  }
  
  // check if all settings are OK
  if(getSettings()){
	// reset watchdog if OK
	resetAlarmState(2);
  }
}
